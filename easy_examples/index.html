    Getting Started

    Easy Examples¶

    These sections assume you are using EasyBot, an extension of ScurryPy designed for convenience. All examples in this section run on their own.

    Setup rules

    • Keep sensitive values (like your bot's token) in a .env file instead of hardcoding them.
    • Always place client.run() at the end of your script, after all commands and event handlers are registered.

    Examples vs API Flexibility

    These examples demonstrate one way to build bots using EasyBot. ScurryPy is fully addon-driven. Every behavior you see here is implemented on top of the core API. If you prefer different patterns, you can extend or replace any part of the system.

    Want to see ScurryPy without Addons? Check out Examples without Addons!

    Basic Event¶

    Demonstrates the absolute minimum to respond to an event.

    import scurrypy 
from scurrypy.addons.easy_bot import EasyBot

client = EasyBot(token=TOKEN, application_id=APP_ID)

@client.event("READY")
async def on_ready(bot, event: scurrypy.ReadyEvent):
    bot.logger.log_info("Bot is online!") # uses bot's logger; print is fine too!

client.run()

    Basic Prefix Command¶

    Demonstrates registering and responding to a command prefix. See Migrations for details on how this differs from slash commands!

    import scurrypy 
from scurrypy.addons.easy_bot import EasyBot

client = EasyBot(token=TOKEN, application_id=APP_ID, prefix="!")

@client.prefix("ping")
async def ping_cmd(bot, message: scurrypy.Message):
    await message.send("Pong!")

client.run()

    Basic Slash Command¶

    Demonstrates registering and responding to a slash command. See Migrations for details on how this differs from prefix commands!

    import scurrypy 
from scurrypy.addons.easy_bot import EasyBot

client = EasyBot(token=TOKEN, application_id=APP_ID)

@client.slash_command("hello", "Say hello", guild_ids=GUILD_ID) # specify guild ID for guild command
async def hello(bot, interaction: scurrypy.Interaction):
    await interaction.respond("Hello!")

client.run()

    About guild commands

    Slash commands appear instantly when registered to a guild. Global commands can take up to 1 hour to propagate.

    Interactive Components¶

    Demonstrates interactive components using buttons. See Migrations for details on how this differs from V2 components!

    import scurrypy
from scurrypy.addons.easy_bot import EasyBot
from scurrypy.addons.component_builder import ComponentBuilder as C

client = EasyBot(token=TOKEN, application_id=APP_ID)

@client.slash_command("button_demo", "Show a button", guild_ids=GUILD_ID)
async def button_demo(bot, interaction: scurrypy.Interaction):

    my_btns = C.row(
        C.primary(custom_id='demo_btn', label='Press')
    )

    await interaction.respond(
        scurrypy.MessagePart(
            content="Click me!",
            components=[my_btns]
        )
    )

@client.button("demo_btn")
async def on_button(bot, component: scurrypy.Interaction):
    await component.update("You pressed the button!")

client.run()

    V2 Components¶

    Demonstrates building Discord's V2 components. See Migrations for details on how this differs from action rows!

    import scurrypy
from scurrypy.addons.easy_bot import EasyBot
from scurrypy.addons.component_builder import ComponentBuilder as C

client = EasyBot(token=TOKEN, application_id=APP_ID)

@client.slash_command("profile", "Show a profile card", guild_ids=GUILD_ID)
async def profile(bot, interaction):

    container = scurrypy.ContainerPart(
        components=[
            scurrypy.SectionPart(
                components=[
                    scurrypy.TextDisplay("**SquirrelBot**\nFast. Light. Nut-powered.")
                ],
                accessory=C.primary(custom_id="show_acorns", label="Your Acorns")
            )
        ]
    )

    await interaction.respond(
        scurrypy.MessagePart(
            components=[container]
        ).set_flags(is_components_v2=True)
    )

@client.button("show_acorns")
async def on_edit(bot, interaction):
    container = scurrypy.ContainerPart(
        components=[
            scurrypy.TextDisplay("**Player Profile**\nAcorns collected: **128**")
        ]
    )

    await interaction.update(
        scurrypy.MessagePart(
            components=[container],
        ).set_flags(is_components_v2=True)
    )

client.run()

    Select Components¶

    Demonstrates interactive select components using string select.

    import scurrypy
from scurrypy.addons.easy_bot import EasyBot
from scurrypy.addons.component_builder import ComponentBuilder as C

client = EasyBot(token=TOKEN, application_id=APP_ID)

@client.slash_command("choose", "Choose option", guild_ids=GUILD_ID)
async def choose(bot, interaction: scurrypy.Interaction):
    row = C.row(scurrypy.StringSelect(
        custom_id="chooser",
        options=[
            C.option(label="A", value="A"),
            C.option(label="B", value="B")
        ]
    ))

    await interaction.respond(
        scurrypy.MessagePart(
            content="Pick one:",
            components=[row]
        )
    )

@client.select("chooser")
async def choice_handler(bot, component: scurrypy.Interaction):
    selected = component.context.data.values[0] # event is stored in context
    await component.respond(f"You chose {selected}")

client.run()

    Modal¶

    Demonstrates using the modal feature.

    import scurrypy
from scurrypy.addons.easy_bot import EasyBot
from scurrypy.addons.component_builder import ComponentBuilder as C

client = EasyBot(token=TOKEN, application_id=APP_ID)

@client.slash_command("feedback", "Leave feedback", guild_ids=GUILD_ID)
async def feedback(bot, interaction: scurrypy.Interaction):
    await interaction.respond_modal(
        scurrypy.ModalPart(
            custom_id="feedback_form",
            title="Feedback",
            components=[
                scurrypy.Label(
                    label='Your feedback',
                    component=C.short_text(custom_id="text"),
                    description='Give us something juicy!'
                )
            ]
        )
    )

@client.modal("feedback_form")
async def submit_feedback(bot, modal: scurrypy.Interaction):
    event: scurrypy.InteractionEvent = modal.context

    text = event.data.get_modal_data('text')

    await modal.respond(f"Thanks for your feedback: {text}")

client.run()

    About modal components

    ScurryPy simplifies Discord's modal structure. Instead of wrapping TextInputs inside ActionRows, ScurryPy uses list[Label] as a clean container for TextInput fields.

    Setting Embeds¶

    Demonstrate building an embed in response to a slash command. See Migrations for details on how this differs from action rows!

    import scurrypy
from scurrypy.addons.easy_bot import EasyBot
from scurrypy.addons.embed_builder import EmbedBuilder as E

client = EasyBot(token=TOKEN, application_id=APP_ID)

@client.slash_command("embedtest", "Send an embed", guild_ids=GUILD_ID)
async def embedtest(bot, interaction: scurrypy.Interaction):
    embed = scurrypy.EmbedPart(
        title="Demo", 
        author=E.user_author(interaction.context.member.user),
        description="Hello!", 
        color=0x00ff99,
        fields=[
            E.field(name='My Field', value='My first field!'),
            E.field(name='Another Field', value='My second field!')
        ],
        timestamp=E.timestamp()
    )
    await interaction.respond(scurrypy.MessagePart(embeds=[embed]))

client.run()

    Stateful Bot¶

    Demonstrates storing user data without a database.

    import scurrypy
from scurrypy.addons.easy_bot import EasyBot
import asyncio

client = EasyBot(token=TOKEN, application_id=APP_ID)

user_points = {}
dict_lock = asyncio.Lock()

@client.slash_command("points", "Check your points", guild_ids=GUILD_ID)
async def points(bot, interaction: scurrypy.Interaction):
    user = interaction.context.member.user.id
    pts = user_points.get(user, 0)
    await interaction.respond(f"You have {pts} points!")

@client.slash_command("addpoints", "Give points", guild_ids=GUILD_ID)
async def addpoints(bot, interaction: scurrypy.Interaction):
    user = interaction.context.member.user.id

    async with dict_lock:
        user_points[user] = user_points.get(user, 0) + 1

    await interaction.respond("Point added!")

client.run()

    asyncio.Lock

    The Lock is required to access user_points to prevent race conditions.

    Autocomplete Interactions¶

    Demonstrates autocompleting a command option's value.

    import scurrypy 
from scurrypy.addons.easy_bot import EasyBot

client = EasyBot(token=TOKEN, application_id=APP_ID)

# A dataset to filter against
CITIES = ["New York", "London", "Tokyo", "Berlin", "Sydney", "Toronto"]

@client.slash_command(
    "travel",
    "Pick a city",
    guild_ids=GUILD_ID,
    options=[
        scurrypy.CommandOption(
            type=scurrypy.CommandOptionTypes.STRING,
            name="city",
            description="Where do you want to go?",
            autocomplete=True
        )
    ]
)
async def travel(bot, interaction: scurrypy.Interaction):
    event: scurrypy.InteractionEvent = interaction.context

    city = event.data.get_option("city")
    await interaction.respond(f"You selected **{city}**!")

@client.autocomplete("travel", "city")
async def travel_city_autocomplete(bot, interaction: scurrypy.Interaction):
    event: scurrypy.InteractionEvent = interaction.context

    # what the user has typed so far
    focused_value = event.data.get_focused_value()

    # filter the cities
    filtered = [
        scurrypy.CommandOptionChoice(name=c, value=c)
        for c in CITIES
        if focused_value.lower() in c.lower()
    ]

    await interaction.respond_autocomplete(filtered)

client.run()

    Deferred Example¶

    Demonstrates using deferred responses to compute long running tasks (both new and update).

    import scurrypy
from scurrypy.addons.easy_bot import EasyBot
import asyncio

client = EasyBot(token=TOKEN, application_id=APP_ID)

@client.slash_command("slowtask_followup", "Deferred with followup", guild_ids=GUILD_ID)
async def slowtask_followup(bot, interaction: scurrypy.Interaction):
    # defer response to avoid 3s timeout
    await interaction.defer_respond(ephemeral=False)  # thinking message visible to everyone

    # simulate a long operation
    await asyncio.sleep(5)

    # send a NEW message as followup
    await interaction.followup(APP_ID, "Task completed! Here's a followup message.")

@client.slash_command("slowtask_edit", "Deferred with edit original", guild_ids=GUILD_ID)
async def slowtask_edit(bot, interaction: scurrypy.Interaction):
    # defer response (thinking message)
    await interaction.defer_respond(ephemeral=True)  # only visible to user

    # simulate long operation
    await asyncio.sleep(5)

    # edit the original deferred message instead of creating a new one
    await interaction.edit_original(APP_ID, "Task completed! Ready for another?")

client.run()

    Pagination¶

    Demonstrates page navigation on an embed.

    import scurrypy
from scurrypy.addons.easy_bot import EasyBot
from scurrypy.addons.component_builder import ComponentBuilder as C
from scurrypy.addons.embed_builder import EmbedBuilder as E

client = EasyBot(token=TOKEN, application_id=APP_ID)

WORD_DICT_INFO = [
    ("Squirrel (noun)", "an agile tree-dwelling rodent with a bushy tail, typically feeding on nuts and seeds"),
    ("acorn (noun)", "the fruit of the oak tree, a smooth oval nut in a rough cup-shaped base"),
    ("toe beans (noun)", "an affectionate term for the soft digital pads on a cat's paws"),
    ("snoot (noun)", "a cutesy, informal term for a dog's nose or snout")
]

WORD_DICT_SIZE = len(WORD_DICT_INFO)

def build_page_buttons(page: int):
    if page == 0:
        start = C.secondary(custom_id='start:0', emoji='', disabled=True)
    else: 
        start = C.primary(custom_id='start:0', emoji='')

    if page -1 < 0:
        back = C.secondary(custom_id=f"back:{page -1}", emoji='', disabled=True)
    else:
        back = C.primary(custom_id=f"back:{page -1}", emoji='')

    if page +1 == WORD_DICT_SIZE:
        next = C.secondary(custom_id=f"next:{page +1}", emoji='', disabled=True)
    else:
        next = C.primary(custom_id=f"next:{page +1}", emoji='')

    if page == WORD_DICT_SIZE -1:
        end = C.secondary(custom_id=f"end:{WORD_DICT_SIZE -1}", emoji='', disabled=True)
    else:
        end = C.primary(custom_id=f"end:{WORD_DICT_SIZE -1}", emoji='')

    # line up buttons in order
    return C.row([start, back, next, end])

def build_dictionary(page: int):
    term, definition = WORD_DICT_INFO[page]

    return scurrypy.MessagePart(
        embeds=[
            scurrypy.EmbedPart(
                color=0xFFD27F,
                title=f'Dictionary',
                fields=[
                    E.field(name=term, value=definition)
                ],
                footer=E.footer(f"Page {page +1} of {WORD_DICT_SIZE}")
            )
        ],
        components=[build_page_buttons(page)]
    )

def extract_page(interaction):
    return int(interaction.context.data.custom_id.split(':')[1])

@client.slash_command('dictionary', 'Open the dictionary!', guild_ids=GUILD_ID)
async def on_dictionary(bot, interaction):
    await interaction.respond(build_dictionary(0))

@client.button('start:0')
async def on_to_start(bot, interaction):
    await interaction.update(build_dictionary(0))

@client.button('back:*')
async def on_back_page(bot, interaction):
    await interaction.update(build_dictionary(extract_page(interaction)))

@client.button('next:*')
async def on_next_page(bot, interaction):
    await interaction.update(build_dictionary(extract_page(interaction)))

@client.button(f"end:{WORD_DICT_SIZE -1}")
async def on_to_end(bot, interaction):
    await interaction.update(build_dictionary(WORD_DICT_SIZE -1))

client.run()
